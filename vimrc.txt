" statusline "{{{
" %n: バッファ番号
" %f: ファイル名
" %m: 修正フラグ[+]/[-]
" %r: 読み込み専用フラグ[RO]
" %w: プレビューウィンドウフラグ[Preview]
" %=: 左寄せと右寄せの区切り
" %c: カラム番号
" %V: 画面上のヴァーチャルカラム番号: -{num} %cと等しいと表示されない
" %l: カーソル行番号
" %L: バッファ総行数
" %P: カーソル行の%表示
" %[1-9]* %*: 強調表示グループを User{N} に設定する "}}}
function! FileStatus() "{{{
    let str = ''
    let str = str . '[' . &fileformat . ']'
    if has('multi_byte') && &fileencoding != ''
        let str = str . '[' . &fileencoding . ']'
    endif
    let str = str . '<' . &filetype . '>'
    return str
endfunction "}}}

" '('で現在カーソルの関数の先頭をプレビュー
" http://subtech.g.hatena.ne.jp/motemen/?word=*[Vim]
function! PreviewOpenBrace() "{{{
    let l:pos = getpos('.')
    if l:pos == get(s:, 'last_pos', [])
        let l:count = s:last_count + 1
    else
        let l:count = v:count1
    endif
    pclose
    pedit
    wincmd p
    execute 'normal!' l:count . '[{'
    setlocal cursorline
    wincmd p
    let s:last_count = l:count
    let s:last_pos   = l:pos
endfunction "}}}
nnoremap <buffer> <silent> ( :<C-U>call PreviewOpenBrace()<CR>

" select tabpage by tab number
function! TabList() "{{{
    " require +dialog_con
    if has('dialog_con') != 1
        call EchoErr("require +dialog_con")
        return
    endif

    " if no tabpages, don't work
    let tabpages = tabpagenr('$')
    if tabpages == 1
    	echo "No tabpages."
    	return
    endif

    " show tab list by 'tabs' command
    let list = ''
    " redirect messages to variable
    redir => list
    tabs
    redir END
    echo "\n"

    " input tab number
    echohl Bold
    let choice = input("Select Tab Number: ")
    echohl None
    "echo "choice: " . choice

    " select tabpage
    if 0 < choice && choice <= tabpages
        execute "tabnext " . choice
    "else
    "    call EchoErr("invalid Number: " . choice)
    endif
endfunction "}}}
nnoremap <silent> T :<C-u>call TabList()<CR>

" vim scouter
function! Scouter(file, ...) "{{{
    let pat = '^\s*$\|^\s*"'
    let lines = readfile(a:file)
    if !a:0 || !a:1
        let lines = split(substitute(join(lines, "\n"), '\n\s*\\', '', 'g'), "\n")
    endif
    return 'Your Vim Power: ' . len(filter(lines,'v:val !~ pat'))
endfunction "}}}

\   echo Scouter(empty(<q-args>) ? $MYVIMRC : expand(<q-args>), <bang>0)

" 新しいファイル名で保存する(rename:古いファイル名は削除される)
function! DoRename(newname) "{{{
    " 0: 偽 それ以外(1):真
    if filereadable(a:newname)
        echoerr "ファイルが存在します: " . a:newname
        return
    endif
    execute 'file ' . a:newname
    execute 'write ' . a:newname
    call delete(expand('#'))
endfunction "}}}
command! -nargs=1 -complete=file Rename :call DoRename(<f-args>)

" Convert tcsh -> zsh
function! DoConvZsh() "{{{
  " 検索なしのエラーを出さない: catchしたらそれ以降やってくれない
  try
    %s/tcsh -f/zsh -f/g
  catch /E486:/
  endtry
  try
    %s/set //g
  catch /E486:/
  endtry
  try
    %s/ = /=/g
  catch /E486:/
  endtry
  try
    %s/\" == \"/\" = \"/g
  catch /E486:/
  endtry
  try
    %s/if (/if [[/g
  catch /E486:/
  endtry
  try
    %s/) then/]] then/g
  catch /E486:/
  endtry
  try
    %s/else if/elif/g
  catch /E486:/
  endtry
  try
    %s/endif/fi/g
  catch /E486:/
  endtry
  try
    %s/goto/#TODO goto/g
  catch /E486:/
  endtry
  try
    %s/^[A-Z].*:$/#TODO function XXX() {/g
  catch /E486:/
  endtry
  try
    %s/unalias \*//g
    delete
  catch /E486:/
  endtry
  set filetype=zsh
endfunction "}}}
command! ConvZsh :<C-u>call DoConvZsh()

"   それぞれ、左から順に呼ばれたことを示す。
"   "vim" でファイルを指定せず起動したとき
"   BufEnter
"   "vim hoge" で存在しないファイルを指定して起動したとき
"   BufNewFile BufEnter
"   "vim hoge" で存在するファイルを指定して起動したとき
"   BufReadPre BufRead BufReadPost BufEnter
"   標準入力から起動したとき
"   StdinReadPre StdinReadPost BufEnter BufEnter
"   "e hoge" で存在しないファイルを開いたとき
"   BufNew BufAdd BufLeave BufHidden BufNewFile BufEnter

"   BufNew BufAdd BufLeave BufHidden BufReadPre BufRead BufReadPost BufEnter
"   "bn" で別のバッファに移動したとき
"   BufLeave BufHidden BufEnter
"   "bd" でバッファを閉じて別のバッファに移動したとき
"   BufLeave BufUnload BufDelete BufEnter
"   "bd" で最後のバッファを閉じたとき
"   BufNew BufAdd BufLeave BufUnload BufEnter BufDelete
"   "w" でファイルに書き出ししたとき
"   BufWrite BufWritePre BufWritePost
"   ":w hoge" で存在しないファイルhogeに書き出ししたとき
"   BufNew BufWrite BufWritePre BufWritePost

function! s:get_syn_id(transparent) "{{{
    let synid = synID(line("."), col("."), 1)
    if a:transparent
        return synIDtrans(synid)
    else
        return synid
    endif
endfunction
function! s:get_syn_attr(synid)
    let name = synIDattr(a:synid, "name")
    let ctermfg = synIDattr(a:synid, "fg", "cterm")
    let ctermbg = synIDattr(a:synid, "bg", "cterm")
    let guifg = synIDattr(a:synid, "fg", "gui")
    let guibg = synIDattr(a:synid, "bg", "gui")
    return {
        \ "name": name,
        \ "ctermfg": ctermfg,
        \ "ctermbg": ctermbg,
        \ "guifg": guifg,
        \ "guibg": guibg}
endfunction
function! s:get_syn_info()
    let baseSyn = s:get_syn_attr(s:get_syn_id(0))
    echo "name: " . baseSyn.name .
        \ " ctermfg: " . baseSyn.ctermfg .
        \ " ctermbg: " . baseSyn.ctermbg .
        \ " guifg: " . baseSyn.guifg .
        \ " guibg: " . baseSyn.guibg
    let linkedSyn = s:get_syn_attr(s:get_syn_id(1))
    echo "link to"
    echo "name: " . linkedSyn.name .
        \ " ctermfg: " . linkedSyn.ctermfg .
        \ " ctermbg: " . linkedSyn.ctermbg .
        \ " guifg: " . linkedSyn.guifg .
        \ " guibg: " . linkedSyn.guibg
endfunction "}}}
command! SyntaxInfo call s:get_syn_info()
